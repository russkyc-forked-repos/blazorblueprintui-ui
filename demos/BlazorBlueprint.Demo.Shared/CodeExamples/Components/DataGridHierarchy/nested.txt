<BbDataGrid TData="MedalEntry" Items="@continents"
            ChildrenSelector="@(c => c.Children)"
            ItemValueSelector="@(c => c.Name)"
            ChildPageSize="5">
    <Columns>
        <BbDataGridHierarchyColumn TData="MedalEntry" TProp="string"
            Property="@(c => c.Name)" Title="Name" Sortable />
        <BbDataGridPropertyColumn TData="MedalEntry" TProp="int"
            Property="@(c => c.Gold)" Title="Gold" Sortable />
        <BbDataGridPropertyColumn TData="MedalEntry" TProp="int"
            Property="@(c => c.Silver)" Title="Silver" Sortable />
        <BbDataGridPropertyColumn TData="MedalEntry" TProp="int"
            Property="@(c => c.Bronze)" Title="Bronze" Sortable />
        <BbDataGridPropertyColumn TData="MedalEntry" TProp="int"
            Property="@(c => c.Total)" Title="Total" Sortable />
    </Columns>
</BbDataGrid>

@code {
    private List<MedalEntry> continents = new()
    {
        new MedalEntry("Asia", 100, 90, 80, new List<MedalEntry>
        {
            new("Japan", 27, 14, 17),
            new("China", 38, 32, 18),
            // ...
        }),
        // ...
    };

    public class MedalEntry
    {
        public string Name { get; set; }
        public int Gold { get; set; }
        public int Silver { get; set; }
        public int Bronze { get; set; }
        public int Total => Gold + Silver + Bronze;
        public List<MedalEntry>? Children { get; set; }
    }
}
