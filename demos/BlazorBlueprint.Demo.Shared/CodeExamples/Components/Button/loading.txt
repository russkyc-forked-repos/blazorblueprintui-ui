<BbButton Loading="@_isLoading" OnClick="SimulateLoading">
    @(_isLoading ? "Submitting..." : "Submit")
</BbButton>

@code {
    private bool _isLoading;

    private async Task SimulateLoading(MouseEventArgs e)
    {
        _isLoading = true;
        StateHasChanged();
        await Task.Delay(2000);
        _isLoading = false;
    }
}
