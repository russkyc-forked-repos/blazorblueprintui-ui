<BbDateRangePicker @bind-Value="_customPresets" Presets="_fullyCustomPresets" />

@code {
    private DateRange? _customPresets;

    private readonly List<DateRangeQuickPick> _fullyCustomPresets = new()
    {
        DateRangeQuickPick.Custom("Last Quarter", () =>
        {
            var today = DateTime.Today;
            var quarterStart = new DateTime(today.Year, ((today.Month - 1) / 3) * 3 + 1, 1).AddMonths(-3);
            return new DateRange(quarterStart, quarterStart.AddMonths(3).AddDays(-1));
        }),
        DateRangeQuickPick.Custom("Next 14 Days", () =>
            new DateRange(DateTime.Today, DateTime.Today.AddDays(13))),
        DateRangeQuickPick.Custom("Year to Date", () =>
            new DateRange(new DateTime(DateTime.Today.Year, 1, 1), DateTime.Today)),
    };
}
