// Value = Label: use SelectOption.FromValue
private static readonly string[] Fruits = ["Apple", "Banana", "Blueberry", "Grapes", "Pineapple"];
private readonly SelectOption<string>[] fruitOptions =
    Fruits.Select(SelectOption.FromValue).ToArray();

<BbSelect TValue="string"
        @bind-Value="selectedFruit"
        Options="@fruitOptions"
        Placeholder="Select a fruit"
        Class="w-[280px]" />
