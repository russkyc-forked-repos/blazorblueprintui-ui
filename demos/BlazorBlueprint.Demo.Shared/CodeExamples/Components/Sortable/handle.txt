@* Set HandleClass to restrict dragging to a specific element inside each item *@
<BbSortable T="MyItem"
            Items="_items"
            ZoneIdentifier="handle-list"
            HandleClass="drag-handle"
            OnItemDropped="HandleReorder">
    <ItemTemplate Context="item">
        <div class="flex items-center gap-3 p-3 rounded-lg border bg-card">
            @* Only this element initiates drag *@
            <span class="drag-handle cursor-grab text-muted-foreground" aria-hidden="true">⠿</span>
            <span class="text-sm font-medium">@item.Title</span>
        </div>
    </ItemTemplate>
</BbSortable>

@code {
    private record MyItem(string Title);
    private List<MyItem> _items = [new("Item A"), new("Item B"), new("Item C")];

    private void HandleReorder(DropItemInfo<MyItem> info)
    {
        if (info.Item is null) return;
        _items.Remove(info.Item);
        _items.Insert(Math.Clamp(info.TargetIndex, 0, _items.Count), info.Item);
    }
}
