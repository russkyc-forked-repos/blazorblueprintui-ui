@code {
    private async ValueTask<DataGridResult<Person>> LoadPeopleAsync(
        DataGridRequest request)
    {
        // Your server/API call here — only the requested window is fetched
        var result = await MyApi.GetPeopleAsync(
            skip: request.StartIndex,
            take: request.Count ?? 50,
            sort: request.SortDefinitions,
            cancellationToken: request.CancellationToken);

        return new DataGridResult<Person>
        {
            Items = result.Items,
            TotalItemCount = result.TotalCount
        };
    }
}
