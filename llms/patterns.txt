# BlazorUI Common Patterns & Multi-Component Examples

This guide shows real-world patterns for using multiple BlazorUI components together.

---

## Form with Validation

Combine Field, Label, Input, and error handling for accessible forms:

```razor
@using BlazorUI.Components

<EditForm Model="@model" OnValidSubmit="@HandleSubmit">
    <DataAnnotationsValidator />

    <Field>
        <FieldLabel>Email</FieldLabel>
        <FieldContent>
            <Input @bind-Value="model.Email" Type="email" Placeholder="name@example.com" />
        </FieldContent>
        <FieldDescription>We'll never share your email with anyone else.</FieldDescription>
        <FieldError>
            <ValidationMessage For="@(() => model.Email)" />
        </FieldError>
    </Field>

    <Field>
        <FieldLabel>Password</FieldLabel>
        <FieldContent>
            <Input @bind-Value="model.Password" Type="password" />
        </FieldContent>
        <FieldDescription>Must be at least 8 characters long.</FieldDescription>
        <FieldError>
            <ValidationMessage For="@(() => model.Password)" />
        </FieldError>
    </Field>

    <div class="flex items-center space-x-2">
        <Checkbox Id="remember" @bind-Checked="model.RememberMe" />
        <Label For="remember">Remember me</Label>
    </div>

    <Button Type="submit" Class="w-full">Sign In</Button>
</EditForm>

@code {
    private LoginModel model = new();

    private async Task HandleSubmit()
    {
        // Handle form submission
    }

    public class LoginModel
    {
        [Required]
        [EmailAddress]
        public string Email { get; set; } = "";

        [Required]
        [MinLength(8)]
        public string Password { get; set; } = "";

        public bool RememberMe { get; set; }
    }
}
```

## Data Table with Sorting and Pagination

Build powerful data tables with sorting, filtering, and pagination:

```razor
@using BlazorUI.Components

<div class="space-y-4">
    <div class="flex items-center justify-between">
        <Input
            @bind-Value="searchTerm"
            Placeholder="Search users..."
            Class="max-w-sm"
            @oninput="FilterUsers" />

        <Button OnClick="AddUser">
            <LucideIcon Name="plus" Size="16" />
            Add User
        </Button>
    </div>

    <DataTable Data="@filteredUsers" @bind-State="tableState">
        <DataTableColumn TData="User" Property="u => u.Name" Title="Name" Sortable="true" />
        <DataTableColumn TData="User" Property="u => u.Email" Title="Email" Sortable="true" />
        <DataTableColumn TData="User" Property="u => u.Role" Title="Role" Filterable="true">
            <Template Context="user">
                <Badge Variant="@GetRoleBadgeVariant(user.Role)">
                    @user.Role
                </Badge>
            </Template>
        </DataTableColumn>
        <DataTableColumn TData="User" Property="u => u.Status" Title="Status">
            <Template Context="user">
                <div class="flex items-center space-x-2">
                    <span class="@GetStatusColor(user.Status)"></span>
                    <span>@user.Status</span>
                </div>
            </Template>
        </DataTableColumn>
        <DataTableColumn TData="User" Title="Actions">
            <Template Context="user">
                <DropdownMenu>
                    <DropdownMenuTrigger AsChild>
                        <Button Variant="ghost" Size="icon">
                            <LucideIcon Name="more-horizontal" Size="16" />
                        </Button>
                    </DropdownMenuTrigger>
                    <DropdownMenuContent>
                        <DropdownMenuItem OnClick="() => EditUser(user)">
                            <LucideIcon Name="edit" Size="16" />
                            Edit
                        </DropdownMenuItem>
                        <DropdownMenuItem OnClick="() => DeleteUser(user)">
                            <LucideIcon Name="trash" Size="16" />
                            Delete
                        </DropdownMenuItem>
                    </DropdownMenuContent>
                </DropdownMenu>
            </Template>
        </DataTableColumn>
    </DataTable>

    <TablePagination State="@tableState" />
</div>

@code {
    private List<User> users = new();
    private List<User> filteredUsers = new();
    private string searchTerm = "";
    private TableState tableState = new();

    protected override void OnInitialized()
    {
        users = GetUsers();
        filteredUsers = users;
    }

    private void FilterUsers(ChangeEventArgs e)
    {
        searchTerm = e.Value?.ToString() ?? "";
        filteredUsers = users.Where(u =>
            u.Name.Contains(searchTerm, StringComparison.OrdinalIgnoreCase) ||
            u.Email.Contains(searchTerm, StringComparison.OrdinalIgnoreCase)
        ).ToList();
    }

    private string GetRoleBadgeVariant(string role) => role switch
    {
        "Admin" => "destructive",
        "Manager" => "default",
        _ => "secondary"
    };
}
```

```razor

<SidebarProvider>

                    <SidebarMenuGroupLabel>Settings</SidebarMenuGroupLabel>

                            <span>Security</span>
                        </SidebarMenuButton>
                    </SidebarMenuItem>
                </SidebarMenuGroup>
            </SidebarMenu>
        </SidebarContent>

        <SidebarFooter>
            <SidebarMenu>
                <SidebarMenuItem>
                    <DropdownMenu>
                        <DropdownMenuTrigger AsChild>
                            <SidebarMenuButton>
                                <Avatar>
                                    <AvatarImage Src="@currentUser.AvatarUrl" />

                                <LucideIcon Name="user" Size="16" />
                                Profile
                            </DropdownMenuItem>
                            <DropdownMenuItem>
                                <LucideIcon Name="settings" Size="16" />

                                Sign Out
                            </DropdownMenuItem>
                        </DropdownMenuContent>
                    </DropdownMenu>

        </SidebarFooter>
    </Sidebar>

        <header class="flex h-16 items-center gap-2 border-b px-4">
            <SidebarTrigger />
            <Separator Orientation="vertical" Class="h-6" />
            <h1 class="text-xl font-semibold">@pageTitle</h1>

    {
        // Handle sign out
```

## Multi-Select with Popover

Create a multi-select dropdown using Popover and Checkbox:

```razor
@using BlazorUI.Components

<Popover>
    <PopoverTrigger AsChild>
        <Button Variant="outline" Class="w-64 justify-between">
            @GetSelectedText()
            <LucideIcon Name="chevron-down" Size="16" />
        </Button>
    </PopoverTrigger>
    <PopoverContent Class="w-64 p-0">
        <Command>
            <CommandInput Placeholder="Search..." />
            <CommandList>
                <CommandEmpty>No results found.</CommandEmpty>
                <CommandGroup>
                    @foreach (var option in options)
                    {
                        <CommandItem OnSelect="() => ToggleOption(option)">
                            <Checkbox
                                Checked="selectedOptions.Contains(option)"
                                Class="mr-2" />
                            @option.Label
                        </CommandItem>
                    }
                </CommandGroup>
            </CommandList>
        </Command>
    </PopoverContent>
</Popover>

@code {
    private List<Option> options = new()
    {
        new("React", "react"),
        new("Vue", "vue"),
        new("Angular", "angular"),
        new("Svelte", "svelte"),
        new("Blazor", "blazor")
    };

    private HashSet<Option> selectedOptions = new();

    private void ToggleOption(Option option)
    {
        if (selectedOptions.Contains(option))
            selectedOptions.Remove(option);
        else
            selectedOptions.Add(option);
    }

    private string GetSelectedText()
    {
        return selectedOptions.Any()
            ? string.Join(", ", selectedOptions.Select(o => o.Label))
            : "Select frameworks...";
    }

    record Option(string Label, string Value);
}
```
